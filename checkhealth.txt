# Añadir clave de Oracle
wget -O- https://www.virtualbox.org/download/oracle_vbox_2016.asc | sudo gpg --yes --dearmor -o /usr/share/keyrings/oracle-virtualbox-2016.gpg

# Añadir repositorio
echo "deb [arch=amd64 signed-by=/usr/share/keyrings/oracle-virtualbox-2016.gpg] https://download.virtualbox.org/virtualbox/debian trixie contrib" | sudo tee /etc/apt/sources.list.d/virtualbox.list

# Actualizar
sudo apt update

# Instalar VirtualBox
sudo apt install -y virtualbox

# (MUY IMPORTANTE) dependencias del kernel
sudo apt install -y build-essential dkms linux-headers-$(uname -r)

# Añadir usuario al grupo
sudo usermod -aG vboxusers $USER

# Reiniciar el sistema
sudo reboot
