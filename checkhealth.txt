#### Terminal: Kitty

```
sudo apt update
sudo apt install -y curl kitty
mkdir -p ~/.config/kitty
curl -LO --output-dir ~/.config/kitty https://raw.githubusercontent.com/catppuccin/kitty/main/themes/mocha.conf
echo "include mocha.conf" >> ~/.config/kitty/kitty.conf
sudo update-alternatives --config x-terminal-emulator
```

#### Fuente: JetBrainsMono

```bash
mkdir -p ~/.local/share/fonts/JetBrainsMono
curl -LO --output-dir ~/.local/share/fonts/JetBrainsMono https://github.com/ryanoasis/nerd-fonts/releases/latest/download/JetBrainsMono.tar.xz
```

#### Shell: Zsh

```bash
# Instalar zsh y plugins
sudo apt update
sudo apt install -y zsh zsh-autosuggestions zsh-syntax-highlighting

# Establecer zsh como shell por defecto
chsh -s $(which zsh)

# Instalar Oh My Zsh
sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/install.sh)"

# Instalar Powerlevel10k
git clone --depth=1 https://github.com/romkatv/powerlevel10k.git \
  ${ZSH_CUSTOM:-$HOME/.oh-my-zsh/custom}/themes/powerlevel10k

# Activar el tema y los plugins en .zshrc
sed -i 's/ZSH_THEME="robbyrussell"/ZSH_THEME="powerlevel10k\/powerlevel10k"/' ~/.zshrc
sed -i 's/plugins=(git)/plugins=(git zsh-autosuggestions zsh-syntax-highlighting)/' ~/.zshrc

# Añadir los plugins al .zshrc
cat >> ~/.zshrc << 'EOF'
autoload -Uz compinit
compinit
source /usr/share/zsh-autosuggestions/zsh-autosuggestions.zsh
source /usr/share/zsh-syntax-highlighting/zsh-syntax-highlighting.zsh
EOF

# Aplicar cambios
source ~/.zshrc
```
